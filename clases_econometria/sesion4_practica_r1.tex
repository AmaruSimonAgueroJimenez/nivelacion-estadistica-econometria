% Slide 7: Prueba t de Student (I)
\begin{frame}[fragile]
  \frametitle{Prueba t: Comparación de Medias}

  \textbf{Hipótesis:} $H_0: \mu_1 = \mu_2$ vs. $H_a: \mu_1 \neq \mu_2$

  \begin{lstlisting}
# t-test de dos muestras independientes
t.test(salario ~ genero, data=datos)

# Salida:
# t = 2.45, df = 98, p-value = 0.0156
# 95 percent confidence interval: [50.2, 450.8]
# mean in group F: 2000
# mean in group M: 1775

# t-test de muestras pareadas
t.test(datos$salario_2023, datos$salario_2024,
       paired=TRUE)
  \end{lstlisting}

  \small
  \textbf{Interpretación:} Si $p\text{-value} < 0.05$, rechazamos $H_0$.
\end{frame}

% Slide 8: Prueba t Avanzada
\begin{frame}[fragile]
  \frametitle{Prueba t: Opciones Avanzadas}

  \begin{lstlisting}
# Especificar nivel de significancia
t.test(salario ~ genero, data=datos,
       alternative="greater", conf.level=0.95)

# Sin asumir varianzas iguales (Welch)
t.test(salario ~ genero, data=datos, var.equal=FALSE)

# Con varianzas iguales (Student)
t.test(salario ~ genero, data=datos, var.equal=TRUE)

# Prueba de igualdad de varianzas (Levene)
library(car)
leveneTest(salario ~ genero, data=datos)
  \end{lstlisting}

  \small
  \textbf{Nota:} Siempre verificar normalidad y homocedasticidad antes de aplicar t-test.
\end{frame}

% Slide 9: Prueba de Proporciones
\begin{frame}[fragile]
  \frametitle{Prueba de Proporciones}

  \textbf{Pregunta:} ¿Es la proporción de mujeres en la muestra diferente de 0.5?

  \begin{lstlisting}
# Tabla de frecuencias
table(datos$genero)
# Female   Male
#    45     55

# Prueba de proporciones
prop.test(45, 100, p=0.5)

# Salida:
# X-squared = 1, df = 1, p-value = 0.3173
# null hypothesis: true p is equal to 0.5
# 95% confidence interval: [0.35, 0.55]
# sample estimate: p = 0.45
  \end{lstlisting}

  \small
  $H_0: p = 0.5$ vs. $H_a: p \neq 0.5$. No rechazamos $H_0$.
\end{frame}

% Slide 10: Prueba Chi-Cuadrado (I)
\begin{frame}[fragile]
  \frametitle{Prueba Chi-Cuadrado: Independencia}

  \textbf{Pregunta:} ¿Son el género y la región independientes?

  \begin{lstlisting}
# Tabla de contingencia
tabla <- table(datos$genero, datos$region)
tabla
#        Norte  Centro  Sur
# Female     20      15   10
# Male       25      20   10

# Prueba chi-cuadrado
chisq.test(tabla)

# Salida:
# X-squared = 0.65, df = 2, p-value = 0.7232
# No evidence of association
  \end{lstlisting}
\end{frame}

% Slide 11: Chi-Cuadrado Avanzado
\begin{frame}[fragile]
  \frametitle{Chi-Cuadrado: Análisis Detallado}

  \begin{lstlisting}
# Guardar resultados
chi_result <- chisq.test(tabla)

# Frecuencias observadas vs. esperadas
chi_result$observed
chi_result$expected

# Residuos estandarizados
chi_result$residuals

# Contribution to chi-square
(chi_result$observed - chi_result$expected)^2 /
  chi_result$expected
  \end{lstlisting}

  \small
  Los residuos permiten identificar celdas que más contribuyen a rechazar $H_0$.
\end{frame}

% Slide 14: Diagnósticos Básicos
\begin{frame}[fragile]
  \frametitle{Diagnósticos: Validación del Modelo}

  \begin{lstlisting}
# Diagnostic plots
plot(modelo_simple)

# Residuos del modelo
residuos <- residuals(modelo_simple)

# Valores ajustados
ajustados <- fitted(modelo_simple)

# Pruebas de normalidad
shapiro.test(residuos)

# Homocedasticidad
library(lmtest)
bptest(modelo_simple)  # Breusch-Pagan
  \end{lstlisting}

  \small
  \textbf{Supuestos MCO:} Linealidad, normalidad de residuos, homocedasticidad, no autocorrelación.
\end{frame}
