% Slide 3: Regresión Múltiple: Especificación
\begin{frame}[fragile]
  \frametitle{Regresión Múltiple (I): Modelo y Estimación}

  \textbf{Modelo:} $Y = \beta_0 + \beta_1 X_1 + \beta_2 X_2 + \cdots + \beta_k X_k + u$

  \begin{lstlisting}
# Regresión múltiple simple
modelo1 <- lm(salario ~ educacion + experiencia,
              data=datos)

# Más regresores
modelo2 <- lm(salario ~ educacion + experiencia +
                edad + genero, data=datos)

# Sintaxis alternativa (todos excepto una variable)
modelo3 <- lm(salario ~ . - id, data=datos)

# Mostrar resultados
summary(modelo1)
summary(modelo2)
  \end{lstlisting}

  \small
  \textbf{Interpretación:} $\beta_j$ mide el cambio en $Y$ cuando $X_j$ aumenta 1 unidad, \textit{ceteris paribus}.
\end{frame}

% Slide 5: Variables Dummy
\begin{frame}[fragile]
  \frametitle{Variables Dummy: Creación y Uso}

  \begin{lstlisting}
# Crear variable dummy manualmente
datos$mujer <- ifelse(datos$genero == "F", 1, 0)
datos$es_profesional <- ifelse(datos$educacion > 12, 1, 0)

# Factor automático (R crea dummies)
datos$genero <- factor(datos$genero)
# R crea: genero1 (Female), genero2 (Male)
# La primera es base, se omite (multicolinealidad)

# Especificar categoría base
datos$region <- factor(datos$region,
                       levels=c("Sur", "Centro", "Norte"))
# Sur es base

modelo_dummy <- lm(salario ~ genero + region, data=datos)
summary(modelo_dummy)
  \end{lstlisting}

  \small
  \textbf{Nota:} R omite automáticamente una categoría para evitar trampa de dummy (multicolinealidad perfecta).
\end{frame}

% Slide 7: Términos de Interacción
\begin{frame}[fragile]
  \frametitle{Interacciones (I): Especificación}

  \textbf{Pregunta:} ¿Difiere el retorno de educación por género?

  \begin{lstlisting}
# Interacción entre educacion y genero
modelo_inter <- lm(salario ~ educacion * genero,
                   data=datos)

# Equivalente a:
modelo_inter2 <- lm(salario ~ educacion + genero +
                           educacion:genero,
                    data=datos)

# Interacción entre dos variables continuas
modelo_inter3 <- lm(salario ~ educacion * experiencia,
                    data=datos)

# Acceder a coeficientes
coef(modelo_inter)
  \end{lstlisting}

  \small
  \texttt{*} expande a efectos principales + interacción; \texttt{:} solo interacción.
\end{frame}

% Slide 9: Diagnóstico: Multicolinealidad
\begin{frame}[fragile]
  \frametitle{Diagnósticos (I): Factor de Inflación de Varianza (VIF)}

  \textbf{Problema:} ¿Las variables explicativas están muy correlacionadas?

  \begin{lstlisting}
library(car)

# VIF para cada regresor
vif(modelo2)
#  educacion  experiencia      edad    generoM
#      1.45       1.82      2.10      1.30

# Regla práctica: VIF > 5-10 sugiere multicolinealidad
# En nuestro caso: todo < 3, sin problemas

# Correlación entre regresores
cor(datos[, c("educacion", "experiencia", "edad")])
  \end{lstlisting}

  \small
  \textbf{VIF bajo (1-2):} No correlacionada con otros regresores\\
  \textbf{VIF alto (>5):} Problema de multicolinealidad
\end{frame}

% Slide 10: Diagnóstico: Normalidad y Homocedasticidad
\begin{frame}[fragile]
  \frametitle{Diagnósticos (II): Validación de Supuestos}

  \begin{lstlisting}
# Gráficos de diagnóstico
par(mfrow=c(2,2))
plot(modelo2)

# Residuos
residuos <- residuals(modelo2)

# Test de normalidad (Shapiro-Wilk)
shapiro.test(residuos)

# Test de homocedasticidad (Breusch-Pagan)
library(lmtest)
bptest(modelo2)

# Test de autocorrelación (Durbin-Watson)
dwtest(modelo2)
  \end{lstlisting}

  \small
  Si p-value \textit{pequeño:} rechazamos supuesto (problema potencial).\\
  Revisar gráficos: residuos vs. ajustados, Q-Q normal.
\end{frame}

% Slide 11: Errores Robustos: Especificación
\begin{frame}[fragile]
  \frametitle{Errores Robustos (I): Heterocedasticidad}

  \textbf{Problema:} ¿La varianza de residuos es constante?

  \begin{lstlisting}
library(sandwich)
library(lmtest)

# MCO estándar
summary(modelo2)

# Errores robustos HC1 (White)
coeftest(modelo2, vcov=vcovHC(modelo2, type="HC1"))

# Errores robustos HC3 (más conservador)
coeftest(modelo2, vcov=vcovHC(modelo2, type="HC3"))

# Alternativa: vcovHAC (heterocedasticidad + autocorrelación)
coeftest(modelo2, vcov=vcovHAC(modelo2))
  \end{lstlisting}

  \small
  \textbf{Nota:} Errores robustos NUNCA aumentan, solo cambian si homocedasticidad falla.
\end{frame}

% Slide 12: Errores Robustos en Tablas
\begin{frame}[fragile]
  \frametitle{Errores Robustos (II): Extracción y Comparación}

  \begin{lstlisting}
# Extraer errores robustos
se_robust <- sqrt(diag(vcovHC(modelo2, type="HC1")))

# Crear tabla comparativa
data.frame(
  Coefficient = coef(modelo2),
  SE_MCO = summary(modelo2)$coefficients[,2],
  SE_Robust = se_robust
)

# Usar stargazer con errores robustos
library(stargazer)
stargazer(modelo2, se=list(se_robust), type="text")
  \end{lstlisting}

  \small
  Comparar errores estándar MCO vs. robustos para detectar homocedasticidad.
\end{frame}

% Slide 15: Flujo Econométrico: Visión General
\begin{frame}
  \frametitle{Flujo Completo del Análisis Econométrico}

  \small
  \begin{enumerate}
    \item \textbf{Formulación teórica:} ¿Qué relación queremos estimar?
    \item \textbf{Especificación del modelo:} $Y = \beta_0 + \beta_1 X_1 + \cdots + u$
    \item \textbf{Recopilación y exploración de datos:} Calidad, valores faltantes, outliers
    \item \textbf{Estimación MCO:} \texttt{lm()} y obtener coeficientes
    \item \textbf{Diagnósticos de supuestos:} Normalidad, homocedasticidad, multicolinealidad
    \item \textbf{Validación de especificación:} ¿Variables correctas? ¿Forma funcional?
    \item \textbf{Correcciones:} Errores robustos, variables adicionales, transformaciones
    \item \textbf{Presentación de resultados:} Tablas, gráficos, interpretación
    \item \textbf{Robustez:} Verificar resultados con otras especificaciones
  \end{enumerate}

  \vspace{0.3cm}
  \small \textit{No es lineal: iteración es normal entre diagnósticos y especificación.}
\end{frame}

% Slide 16: Ejemplo Integrado: Paso a Paso
\begin{frame}[fragile]
  \frametitle{Flujo Práctico: Ejemplo Completo}

  \tiny
  \begin{lstlisting}
datos <- read.csv("datos_laborales.csv"); str(datos)
modelo1 <- lm(log(salario) ~ educacion + experiencia, data=datos)
# Diagnostics
shapiro.test(residuals(modelo1)); bptest(modelo1); vif(modelo1)
# Improved model
modelo2 <- lm(log(salario) ~ educacion*genero +
              experiencia + I(experiencia^2), data=datos)
# Robust errors
coeftest(modelo2, vcov=vcovHC(modelo2, "HC1"))
stargazer(modelo1, modelo2, type="latex")
  \end{lstlisting}
\end{frame}
